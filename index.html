Jalankan Verifikasi Lokasi Sekali Saja saat Dashboard Terbuka checkLocationOnce(); }; function checkLocationOnce() { const config = JSON.parse(localStorage.getItem('masook_config')) || { gpsEnabled: true, lat: -2.395, lng: 116.216, radius: 100 }; const statusDiv = document.getElementById('status-radius'); const statusText = document.getElementById('radius-text'); const btnScan = document.getElementById('btn-scan-main'); if (!config.gpsEnabled) { setValid(statusDiv, statusText, btnScan, "GPS Dinonaktifkan (Mode Bebas)"); return; } // Opsi koordinat agar lebih cepat terkoneksi const geoOptions = { enableHighAccuracy: true, // Meminta akurasi tinggi timeout: 10000, // Maksimal menunggu 10 detik maximumAge: 0 // Jangan gunakan data lokasi lama yang tersimpan di cache }; navigator.geolocation.getCurrentPosition((pos) => { const distance = calculateDistance( pos.coords.latitude, pos.coords.longitude, config.lat, config.lng ); if (distance <= config.radius) { setValid(statusDiv, statusText, btnScan, "📍 Lokasi Valid (Dalam Radius)"); } else { setInvalid(statusDiv, statusText, btnScan, `🚫 Luar Radius (${Math.round(distance)}m)`); } }, (err) => { // Jika error atau timeout, berikan pesan spesifik let errorMsg = "⚠️ GPS Gagal Dideteksi"; if (err.code === 1) errorMsg = "⚠️ Izin Lokasi Ditolak"; if (err.code === 3) errorMsg = "⚠️ GPS Lemah / Timeout"; setInvalid(statusDiv, statusText, btnScan, errorMsg); console.error("Geo Error:", err); }, geoOptions); } function calculateDistance(lat1, lon1, lat2, lon2) { const R = 6371e3; const φ1 = lat1 * Math.PI/180; const φ2 = lat2 * Math.PI/180; const Δφ = (lat2-lat1) * Math.PI/180; const Δλ = (lon2-lon1) * Math.PI/180; const a = Math.sin(Δφ/2) * Math.sin(Δφ/2) + Math.cos(φ1) * Math.cos(φ2) * Math.sin(Δλ/2) * Math.sin(Δλ/2); return R * (2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a))); }
